module IF_ID_reg(
input clk,
input [31:0] PC_out_IF,
input [31:0] Instruction_out_IF,

output [31:0] PC_in_ID,
output [31:0] Instruction_in_ID
);

reg [31:0] PC_reg;
reg [31:0] Instr_reg;

assign PC_in_ID = PC_reg;
assign Instruction_in_ID = Instr_reg;

always @(posedge clk)
begin
PC_reg <= PC_out_IF;
Instr_reg <= Instruction_out_IF;
end

endmodule
